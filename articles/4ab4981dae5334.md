### Wi-Fi

```python
import rp2
import network
import time

ssid     = 'SSID'
password = 'PASSWORD'

def connect(ssid, password):
  rp2.country('JP')
  wlan = network.WLAN(network.STA_IF)
  wlan.active(True)
  wlan.config(pm = 0xa11140)
  wlan.connect(ssid, password)

  while not wlan.isconnected() and 0 <= wlan.status():
    print("wifi connecting...")
    time.sleep(1)

  ip = wlan.ifconfig()[0]
  print(f"IP Address: {ip}")

connect(ssid, password)
```

### NTPで時刻を取得

```python
import rp2
import network
import time
import ntptime

def set_time(ntp_server):
  ntptime.host = ntp_server
  try:
    ntptime.settime()
    time.sleep(1)
    print("NTP 時刻同期に成功しました")
  except:
    print("NTP 時刻同期に失敗しました")
    raise

# 日本標準時(JST = UTC + 9)を取得する
def get_time():
  utc = time.time()
  add_time = 9 * 60 * 60
  t = time.localtime(utc + add_time)
  res = {
    'year': f"{t[0]}",
    'mon' : f"{t[1]:02d}",
    'day' : f"{t[2]:02d}",
    'hour': f"{t[3]:02d}",
    'min' : f"{t[4]:02d}",
    'sec' : f"{t[5]:02d}",
  }
  return res

connect(ssid, password)
set_time("time.google.com")

while True:
  t = get_time()
  print(f"{t.year}/{t.mon}/{t.day} {t.hour}:{t.min}:{t.sec}")
  time.sleep(1)
```

### 実装

```python
import rp2
import network
import time
import ntptime
import tm1637
from Pin import machine

ssid     = 'SSID'
password = 'PASSWORD'
clk = Pin(2)
dio = Pin(0)
tm = im1637.TM1637(clk=clk, dio=dio)

def connect(ssid, password):
  rp2.country('JP')
  wlan = network.WLAN(network.STA_IF)
  wlan.active(True)
  wlan.config(pm = 0xa11140)
  wlan.connect(ssid, password)

  while not wlan.isconnected() and 0 <= wlan.status():
    print("wifi connecting...")
    time.sleep(1)

  ip = wlan.ifconfig()[0]
  print(f"IP Address: {ip}")

def set_time(ntp_server):
  ntptime.host = ntp_server
  try:
    ntptime.settime()
    time.sleep(1)
    print("NTP 時刻同期に成功しました")
  except:
    print("NTP 時刻同期に失敗しました")
    raise

# 日本標準時(JST = UTC + 9)を取得する
def get_time():
  utc = time.time()
  add_time = 9 * 60 * 60
  t = time.localtime(utc + add_time)
  return t

def display(min, sec):
  tm.brightness(0)      # 0-7
  tm.numbers(min, sec)

connect(ssid, password)
set_time("time.google.com")

while True:
  t = get_time()
  display(t[3], t[4])
  time.sleep(1)
```

## TM1637の動作

- [TM1637-LEDをライブラリーなしで簡単表示](https://a-tomi.hatenablog.com/entry/2020/07/25/164923)

- init
  - clk: HIGH
  - dio: HIGH

- start
  DIO を LOW, CLK を LOW
  1. dio: LOW
  2. 50msec wait
  3. clk: LOW
  4. 50msec wait

- stop
  CLK を HIGH, DIO を HIGH
  1. clk: HIGH
  2. 50msec wait
  3. dio: HIGH
  4. 50msec wait

- send
  - 8 ビットを小さいビットから 1 ビットずつ送信
  - 1 ビット送信後にclock_on
  - 最後に、DIO に LOW して clock_on

- clock_on
  CLK を HIGH => LOW
  1. clk: HIGH
  2. 50msec wait
  3. clk: LOW
  4. 50msec wait

- データ送信
  - "0x40"(連続書込モード) を送信
    - start + "0x40" + stop
  - "0xC0"(書込アドレス) の後に4文字分送信
    - start + "0xC0" + "0xXX" + "0xXX" + "0xXX" + "0xXX" + stop
  - "0x8X" と X=8+(明るさ0-7) を送信。X=0にすると消灯する
    - start + "0x88" + stop

  - Data command
    "0100 0000"(40): Write Data + Auto Address + Normal Mode

    | B7  | B6  | B5  | B4  | B3  | B2  | B1  | B0  |     内容     |
    | :-: | :-: | :-: | :-: | :-: | :-: | :-: | :-: | :----------: |
    |  0  |  1  |  0  |  0  |  X  |  X  |  0  |  0  |  Write Data  |
