```js:multicast_udp.js
import dgram from 'node:dgram'
import os from 'os'

// settings ###############################################
const MULTICAST_IP = '225.1.0.0'
const MULTICAST_PORT = 12346
const CMD_GET_INFO = 'send_ip'
let is_debug = false
// ########################################################

// functions ##############################################

function log() {
  if (is_debug) {
    console.log.apply(null, arguments)
  }
}

// udp マルチキャストソケットの初期化
// on_message: メッセージ受信時のコールバック (msg, rinfo) => {}
function init_udp_socket(on_message = () => {}) {
  const socket = dgram.createSocket('udp4')

  socket.on('close', () => {
    log(`socket closed. ${socket.address().address}`)
  })

  socket.on('listening', () => {
    const address = socket.address()
    log(`UDP socket listening on ${address.address}:${address.port}`)

    socket.setBroadcast(true)
    socket.setMulticastTTL(128)
  })

  socket.on('message', (msg, rinfo) => {
    log(`get udp packet msg: [${msg}] from ${rinfo.address}:${rinfo.port}`)
  })
  socket.on('message', on_message)

  return socket
}

// IPアドレスを取得する
// nicname: NIC インターフェイス名
function ip_address(nicname = null) {
  if (nicname === null) {
    nicname = (process.platform === 'win32')? "イーサネット": "en0"
  }

  const nets = os.networkInterfaces()
  const net = nets[nicname]?.find((v) => v.family?.toLowerCase() == "ipv4")
  const res = !!net ? net.address: null
  log(`network interface is ${res}`)
  return res
}
// ########################################################

// UDP でメッセージを送信
// socket: udp socket
// msg: メッセージ
function udp_send(socket, msg) {
  if (typeof msg !== 'string') {
    msg = JSON.stringify(msg)
  }
  socket.send(msg, 0, msg.length, MULTICAST_PORT, MULTICAST_IP)
  log(`send udp packet msg: [${msg}] to ${MULTICAST_IP}:${MULTICAST_PORT}`)
}

// IP お知らせコマンドを送信する
function send_get_info() {
  if (server !== null) {
    udp_send(server, CMD_GET_INFO)
  }
}

// メッセージがコマンドかどうか
function get_msg(msg) {
  msg = msg.toString()
  return (msg === CMD_GET_INFO)? false: JSON.parse(msg)
}

// udp サーバを初期化する
// on_message: クライアントからのメッセージ受信イベント
// debug     : デバッグ表示するかどうか
let server = null
function init_server({ on_message, debug } = { on_message: () => {}, debug: false }) {
  is_debug = debug

  if (server === null) {
    server = init_udp_socket(on_message)
    server.bind(MULTICAST_PORT, () => {
      server.addMembership(MULTICAST_IP)  // マルチキャスト受信を有効にする
    })
  }
  return server
}

// udp クライアントを初期化する
// UDP サーバから、コマンド "CMD_GET_INFO" を受信したらIPを返す
// if_name: IPを取得するインタフェース名
// debug  : デバッグ表示するかどうか
function init_client({ if_name, debug } = { if_name: null, debug: false }) {
  is_debug = debug

  const client = init_udp_socket((msg) => {
    // コマンドを受信したら送信先に内容を返す
    if (msg.toString() === CMD_GET_INFO) {
      const res_msg = {
        ip: ip_address(if_name),
        name: os.hostname(),
      }
      udp_send(client, res_msg)
    }
  })

  client.bind(MULTICAST_PORT, ip, () => {
    client.addMembership(MULTICAST_IP)  // マルチキャスト受信を有効にする
  })
}

export {
  init_server,
  init_client,
  send_get_info,
  get_msg,
  udp_send,
}
```

```js:server.js
import {
  init_server,
  get_msg,
  send_get_info,
} from "./multicast_udp.js"

const debug = false
const interval = 2000

function on_message(msg) {
  // 複数機器からコマンド結果を受信
  msg = get_msg(msg)
  if (msg !== false) {
    console.log("get message:", msg)
  }
}

// main
init_server({
  on_message,
  debug,
})

setTimeout(() => send_get_info(), interval)
```
